# FJStatusBarHUD
一个简单的状态栏指示器

## 显示成功信息
++ (void)fj_showSuccess:(NSString *)msg;

## 显示失败信息
++ (void)fj_showError:(NSString *)msg;

## 显示正在加载信息
++ (void)fj_showLoading:(NSString *)msg;

## 显示普通信息
++ (void)fj_showMessage:(NSString *)msg;

## 显示自定义信息
++ (void)fj_showMessage:(NSString *)msg image:(UIImage *)image;
